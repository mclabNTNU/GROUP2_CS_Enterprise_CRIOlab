1function [Tau,Rz1] = fcn(s_dot,Us,Us_s,eta_d,eta_ds,eta_ds2,eta,nu)
 2
 3%% ROTATION MATRICES
 4
 5psi = 0;
 6Tau = [0;0;0];
 7psi = eta(3);   % psi : yaw angle
 8r = nu(3);      % r : yaw rate
 9u = nu(1);      % u : surge velocity
10v = nu(2);      % v : sway velocity
11
12
13R = [ cos(psi) , -sin(psi) , 0 ;
14      sin(psi) , cos(psi)  , 0 ;
15      0            0         1 ] ;
16
17S = [ 0 , -r , 0 ;
18       r , 0 , 0 ;
19       0 , 0 , 0 ];
20
21   
22%% MASS MATRIX (from CSEI_tau)
23
24m    = 14.79;   I_z  =  1.7600;   x_g = 0.0375;       % x_g  =  0.0460; % Table B.1
25
26% Added mass
27X_ud  =     -2.0;                                           % Table B.1
28                    Y_vd =     -1.0;    N_vd =    -0.0;     % Table B.1 guestimate
29                    Y_rd =      -0.0;   N_rd =    -1.0;     % Table B.1
30                    
31% Total mass matrix
32m_11 = m-X_ud;
33m_22 = m-Y_vd;
34m_23 = m*x_g-Y_rd;
35m_32 = m*x_g-N_vd;
36m_33 = I_z-N_rd;
37
38M    = [m_11 0    0
39        0    m_22 m_23
40        0    m_32 m_33];
41    
42%% DAMPING MATRIX (from CSEI_tau)
43X_u     = -0.6555;      X_v     = 0;
44X_uu    = 0.3545;       X_vv    = -2.443;
45X_uuu   = -3.787;       X_vvv   = 0;
46
47                        Y_v     = -1.33;            Y_r     = -7.250; % Y_r
48                        Y_vv    = -2.776;           Y_rr    = -3.450; % Y_rr
49                        Y_vvv   = -64.91;           Y_rrr   = 0;
50                        
51                        Y_rv    = -0.805;           Y_vr    = -0.845; % Y_rv, Y_vr
52
53                        N_v     = 0;                N_r     = -1.900; % N_r
54                        N_vv    = -0.2088;          N_rr    = -0.750; % N_rr
55                        N_vvv   = 0;                N_rrr   = 0;
56                        
57                        N_rv    = 0.130;            N_vr    = 0.080; % N_rv, N_vr
58
59% Assembly of the damping matrix
60d_11 = -X_u - X_uu*u - X_uuu*u^2;
61d_22 = -Y_v - Y_vv*v - Y_vvv*v^2 - Y_rv*abs(r);
62d_33 = -N_r - N_rr*abs(r) - N_rrr*r^2 - N_vr*abs(v); % using abs(r)
63
64d_12 = -X_v - X_vv*v - X_vvv*v^2;
65d_23 = -Y_r - Y_rr*abs(r) - Y_rrr*r^2 - Y_vr*abs(v);
66d_32 = -N_v - N_vv*v - N_vvv*v^2 - N_rv*abs(r);
67
68D    = [d_11 d_12 0
69        0    d_22 d_23
70        0    d_32 d_33];
71
72
73%% CASE D
74
75
76K_p = diag([0.2,0.2,0.1]);          % Kp (tuning)
77K2 = 0.1 * diag([1,1,1]);           % K2 (tuning)
78
79z1 = R'*(eta-eta_d);                % z1
80alpha1 = -K_p*z1 + R'*eta_ds*Us ;   % alpha1
81Rz1 = R*z1 ;                        % R*z1
82
83z1_dot = S'*z1+nu-R'*eta_ds*s_dot;  % z1_dot
84a1_dot = -K_p*z1_dot+(R*S)'*eta_ds*Us + R'*(eta_ds2*Us+eta_ds*Us_s)*s_dot;    % alpha1_dot
85
86z2 = nu - alpha1 ;                  % z2
87
88Tau = D*nu  +   M*a1_dot - K2*z2 - z1 ;     % Tau
89
90
91
92end
